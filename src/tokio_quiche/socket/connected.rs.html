27use datagram_socket::DatagramSocketRecv;
28use datagram_socket::DatagramSocketSend;
29use std::io;
30use std::net::SocketAddr;
31use std::sync::Arc;
32use tokio::net::UdpSocket;
33
34/// A connected datagram socket with separate `send` and `recv` halves.
35///
36/// [`Socket`] abstracts over both real UDP-based connections and in-process
37/// tunneled flows like (multi-hop) MASQUE flows. It uses the
38/// [`datagram_socket`] traits for this purpose.
39#[derive(Debug)]
40pub struct Socket<Tx, Rx> {
41    /// The sending half of the connection. This generally supports concurrent
42    /// senders.
43    pub send: Tx,
44    /// The receiving half of the connection. This is generally owned by a
45    /// single caller.
46    pub recv: Rx,
47    /// The address of the local endpoint.
48    pub local_addr: SocketAddr,
49    /// The address of the remote endpoint.
50    pub peer_addr: SocketAddr,
51}
52
53/// A type-erased variant of [`Socket`] with boxed `Tx` and `Rx` halves.
54pub type BoxedSocket = Socket<
55    Box<dyn DatagramSocketSend + Send + 'static>,
56    Box<dyn DatagramSocketRecv + Sync + 'static>,
57>;
58
59impl<Tx, Rx> Socket<Tx, Rx> {
60    /// Creates a [`Socket`] from a [`UdpSocket`] by wrapping the file
61    /// descriptor in an [`Arc`].
62    pub fn from_udp(
63        socket: UdpSocket,
64    ) -> io::Result<Socket<Arc<UdpSocket>, Arc<UdpSocket>>> {
65        let local_addr = socket.local_addr()?;
66        let peer_addr = socket.peer_addr()?;
67
68        let send = Arc::new(socket);
69        let recv = Arc::clone(&send);
70
71        Ok(Socket {
72            send,
73            recv,
74            local_addr,
75            peer_addr,
76        })
77    }
78}
79
80impl<Tx, Rx> Socket<Tx, Rx>
81where
82    Tx: DatagramSocketSend,
83    Rx: DatagramSocketRecv,
84{
85    /// Checks whether both `send` and `recv` refer to the same underlying
86    /// UDP socket FD and returns a reference to that socket.
87    ///
88    /// # Note
89    /// The file descriptor _numbers_ have to be identical. A pair of FDs
90    /// created by [`dup(2)`](https://man7.org/linux/man-pages/man2/dup.2.html) will
91    /// return `None`.
92    #[cfg(unix)]
93    pub fn as_udp_socket(&self) -> Option<&UdpSocket> {
94        use std::os::fd::AsRawFd;
95
96        let send = self.send.as_udp_socket()?;
97        let recv = self.recv.as_udp_socket()?;
98        (send.as_raw_fd() == recv.as_raw_fd()).then_some(send)
99    }
100}
101
102impl TryFrom<UdpSocket> for Socket<Arc<UdpSocket>, Arc<UdpSocket>> {
103    type Error = io::Error;
104
105    fn try_from(socket: UdpSocket) -> Result<Self, Self::Error> {
106        Self::from_udp(socket)
107    }
108}
